Experiment-level parameter recovery audit¶

Question: does the current browser adaptive workflow recover known simulated participants, and how do Stan MI ADO, Stan random designs, and Quest+ compare on the same experiment grid?

This notebook drives the actual jsPsych experiment page in simulate=data-only mode. It is an explainer/audit template, not CI and not a definitive efficiency study. The table used for analysis is normalized to one row per experiment, model, strategy label, simulation profile, seed, trial, and parameter so the same notebook shape can cover future models with different parameter counts.

The comparison arms answer slightly different questions. Stan MI ADO is the main workflow: Stan/WASM posterior updates with mutual-information design selection. Stan random keeps the same Stan/WASM posterior updates but samples designs randomly from the same grid, isolating the value of adaptive design selection. Quest+ is a separate discrete-grid adaptive controller and posterior, so it is an alternative-controller benchmark rather than a pure design-strategy ablation.

The rendered reports use fixed randomized settings files so the empirical error bands and true-vs-recovered plots are reproducible. By default, the parameter cell still runs the compact delay-discounting profile audit. To run a rendered report, execute the notebook with papermill and set SETTINGS_PATH to a JSON settings file such as examples/parameter_recovery/delay_discounting_random_recovery_settings.json or examples/parameter_recovery/line_length_discrimination_random_recovery_settings.json.

Final-trial recovery¶

Each row below summarizes the last available trial for one parameter and strategy. Error is computed on the parameter's configured scale: log10 for parameters such as k, raw units for linear parameters. Rank correlation is computed between true values and final posterior means on the same transformed plotting scale.

In [5]:
def transform_value(value, scale):
    if pd.isna(value):
        return math.nan
    value = float(value)
    if scale == "log10":
        return math.log10(value) if value > 0 else math.nan
    return value


def rank_correlation(group, scale):
    transformed = pd.DataFrame({
        "true_plot": [transform_value(value, scale) for value in group["true_value"]],
        "post_plot": [transform_value(value, scale) for value in group["post_mean"]],
    }).dropna()
    if len(transformed) < 2:
        return math.nan
    return transformed["true_plot"].rank().corr(transformed["post_plot"].rank())


final_rows = (
    parameter_rows
    .sort_values("trial_number")
    .groupby(["strategy", "profile_id", "seed", "parameter"], as_index=False)
    .tail(1)
)

summary_rows = []
for (parameter, label, scale, strategy), group in final_rows.groupby(
    ["parameter", "parameter_label", "parameter_scale", "strategy"],
    sort=False,
):
    abs_error = group["abs_error"].astype(float)
    post_sd = group["post_sd"].astype(float)
    summary_rows.append({
        "parameter": parameter,
        "parameter_label": label,
        "parameter_scale": scale,
        "strategy": strategy,
        "median_abs_error": abs_error.median(),
        "q25_abs_error": abs_error.quantile(0.25),
        "q75_abs_error": abs_error.quantile(0.75),
        "iqr_abs_error": abs_error.quantile(0.75) - abs_error.quantile(0.25),
        "median_post_sd": post_sd.median(),
        "rank_correlation": rank_correlation(group, scale),
        "n_runs": len(group),
    })

final_summary = (
    pd.DataFrame(summary_rows)
    .sort_values(["parameter", "strategy"])
    .reset_index(drop=True)
)

Directional checks¶

Some recovery audits need a model-specific sanity check: when one simulation profile has a higher true value for a parameter than another, does the final posterior mean recover the same ordering? These checks are optional and come from the settings JSON.

directional_rows = []
for check in DIRECTIONAL_CHECKS:
    parameter_name = check["parameter"]
    lower_profile = check["lower_profile"]
    higher_profile = check["higher_profile"]

            "higher_post_mean": higher_post_mean,
            "difference": higher_post_mean - lower_post_mean,
            "passed": higher_post_mean > lower_post_mean,

def plot_band_by_parameter(
    data,
    center,
    lower,
    upper,
    ylabel,
    selected_parameters,
    outer_lower=None,
    outer_upper=None,
):
    for parameter_name in selected_parameters:
        plot_data = data[data["parameter"] == parameter_name]
        if plot_data.empty:
            continue
        label = plot_data["parameter_label"].iloc[0]
        fig, ax = plt.subplots(figsize=(7, 4))
        for strategy, strategy_data in plot_data.groupby("strategy"):
            strategy_data = strategy_data.sort_values("trial_number")
            x = strategy_data["trial_number"].to_numpy(dtype=float)
            y = strategy_data[center].to_numpy(dtype=float)
            line = ax.plot(x, y, label=strategy, linewidth=1.8)[0]
            color = line.get_color()
            if outer_lower and outer_upper:
                ax.fill_between(
                    x,
                    strategy_data[outer_lower].to_numpy(dtype=float),
                    strategy_data[outer_upper].to_numpy(dtype=float),
                    color=color,
                    alpha=0.10,
                    linewidth=0,
                )
            ax.fill_between(
                x,
                strategy_data[lower].to_numpy(dtype=float),
                strategy_data[upper].to_numpy(dtype=float),
                color=color,
                alpha=0.22,
                linewidth=0,
            )
        ax.set_title(label)
        ax.set_xlabel("Trial")
        ax.set_ylabel(ylabel)
        ax.legend(title="Strategy")
        ax.grid(alpha=0.25)
        plt.show()

def plot_final_error_distribution(data, selected_parameters):
    for parameter_name in selected_parameters:
        plot_data = data[data["parameter"] == parameter_name]
        if plot_data.empty:
            continue
        label = plot_data["parameter_label"].iloc[0]
        strategies = [strategy for strategy in STRATEGIES if strategy in set(plot_data["strategy"])]
        values_by_strategy = [
            plot_data[plot_data["strategy"] == strategy]["abs_error"].astype(float).tolist()
            for strategy in strategies
        ]
        fig, ax = plt.subplots(figsize=(7, 4))
        positions = list(range(1, len(strategies) + 1))
        ax.boxplot(values_by_strategy, positions=positions, showfliers=False)
        ax.set_xticks(positions)
        ax.set_xticklabels(strategies)
        for position, values in zip(positions, values_by_strategy):
            offsets = [((index % 9) - 4) * 0.018 for index in range(len(values))]
            ax.scatter(
                [position + offset for offset in offsets],
                values,
                color="black",
                alpha=0.18,
                s=9,
                linewidths=0,
            )
        ax.set_title(label)
        ax.set_xlabel("Strategy")
        ax.set_ylabel("Final absolute error")
        ax.grid(axis="y", alpha=0.25)
        plt.show()


def plot_true_vs_recovered(data, selected_parameters):
    for parameter_name in selected_parameters:
        plot_data = data[data["parameter"] == parameter_name].copy()
        if plot_data.empty:
            continue
        label = plot_data["parameter_label"].iloc[0]
        scale = plot_data["parameter_scale"].iloc[0]
        plot_data["true_plot"] = [transform_value(value, scale) for value in plot_data["true_value"]]
        plot_data["post_plot"] = [transform_value(value, scale) for value in plot_data["post_mean"]]
        plot_data = plot_data.dropna(subset=["true_plot", "post_plot"])
        if plot_data.empty:
            continue
        fig, ax = plt.subplots(figsize=(5.5, 5))
        for strategy, strategy_data in plot_data.groupby("strategy"):
            ax.scatter(
                strategy_data["true_plot"],
                strategy_data["post_plot"],
                alpha=0.45,
                s=18,
                label=strategy,
            )
        axis_min = min(plot_data["true_plot"].min(), plot_data["post_plot"].min())
        axis_max = max(plot_data["true_plot"].max(), plot_data["post_plot"].max())
        padding = (axis_max - axis_min) * 0.05 if axis_max > axis_min else 1
        axis_min -= padding
        axis_max += padding
        ax.plot([axis_min, axis_max], [axis_min, axis_max], color="black", linestyle="--", linewidth=1)
        ax.set_xlim(axis_min, axis_max)
        ax.set_ylim(axis_min, axis_max)
        axis_label = f"log10({parameter_name})" if scale == "log10" else parameter_name
        ax.set_title(label)
        ax.set_xlabel(f"True {axis_label}")
        ax.set_ylabel(f"Recovered {axis_label}")
        ax.legend(title="Strategy")
        ax.grid(alpha=0.25)
        plt.show()
